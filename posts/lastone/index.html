      Quick Sort

      def quick(q, l, r):
	if l >= r:
		return
	p, i, j = q[l + r >> 1], l - 1, r + 1
	while i < j:
		while 1:
			i += 1
			if q[i] >= p:
				break
		while 1:
			j -= 1
			if q[j] <= p:
				break
		if i < j:
			q[i], q[j] = q[j], q[i]
	quick(q, l, j)
	quick(q, j + 1, r)
